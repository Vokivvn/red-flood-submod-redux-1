create_pie_chart = {
	clear_array = pie_chart
	resize_array = {
		array = pie_chart
		value = 1 # 1 is value for neutrality frame, so we don't have to make a loop for it
		size = 100 # 100 pieces, each percent is one piece
	}
	set_temp_variable = { keller_pop = FIU_keller_popularity }

	set_temp_variable = { marinetti_pop = FIU_marinetti_popularity }
	set_variable = { ROOT.last_idx = keller_pop }
	add_to_temp_variable = { marinetti_pop = ROOT.last_idx }
	for_loop_effect = {
		start = ROOT.last_idx
		end = marinetti_pop
		set_variable = { pie_chart^v = 2 }
		add_to_variable = { ROOT.last_idx = 1 }
	}

	set_temp_variable = { shimoi_pop = FIU_shimoi_popularity }
	add_to_temp_variable = { shimoi_pop = ROOT.last_idx }
	for_loop_effect = {
		start = ROOT.last_idx
		end = shimoi_pop
		set_variable = { pie_chart^v = 3 }
	}

	clear_variable = ROOT.last_idx
	add_to_variable = { pie_chart_dirty = 1 }
}
