political_actions = {
	return_cores_to_ally = {
		icon = generic_agreement

		available = {
			if = {
				limit = { has_war = yes }
				surrender_progress < 0.20
			}
			else = {
				has_war = no
			}
		}

		target_array = faction_members
		target_array = subjects

		target_trigger = {
			country_exists = FROM
			is_ally_with = FROM
			NOT = {
				has_country_flag = RF_returning_land
				has_country_flag = RF_checking_for_possible_annexations
			}
			any_owned_state = {
				is_claim_or_core_of_FROM = yes
				is_claim_or_core_of_ROOT = no

				NOT = { #Don't hand over our direct colonial territory
					AND = {
						ROOT = { tag = ENG }
						is_british_colonial_claim = yes
					}
					AND = {
						ROOT = { tag = JAP }
						is_japanese_colonial_claim = yes
					}
					AND = {
						ROOT = { tag = ITA }
						is_italian_colonial_claim = yes
					}
				}
				#NOT = { has_variable = can_integrate_@ROOT }
				#NOT = { any_country_with_core = { has_variable = ROOT.can_integrate_@THIS } }
				NOT = { has_state_flag = RF_keeping_state@ROOT }
				OR = {
					is_coastal = yes
					any_neighbor_state = { is_owned_by = FROM }
				}
			}
		}

		cost = 0
		fire_only_once = no

		ai_will_do = {
			factor = 10
			modifier = {
				factor = 0.1
				FROM = {
					is_subject = yes
					NOT = { is_subject_of = ROOT }
				}
			}
		}

		complete_effect = {
			log = "[GetDateText]: [Root.GetName]: Decision return_cores_to_ally target: [From.GetName]"
			set_country_flag = RF_returning_land
			FROM = { save_event_target_as = return_to }
			country_event = rf.political.1
		}
	}
}
