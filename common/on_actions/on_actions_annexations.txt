	on_annex = {
		effect = {
			if = {
				limit = {
					NOT = { has_country_flag = RF_checking_for_possible_annexations }
					is_subject = no
				}
				set_country_flag =  RF_checking_for_possible_annexations
				country_event = { id = annex.1 days = 1 random_days = 3 }
			}
		}
	}

					NOT = { has_country_flag = RF_checking_for_possible_annexations }
