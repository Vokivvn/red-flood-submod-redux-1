country_event = {
	id = ISR.victory
	title = ISR.victory.t
	desc = ISR.victory.desc
	picture = GFX_ISR_flag_raise
	fire_only_once = yes
	
    trigger = {
		tag = ISR
		NOT = {
			ISR = {
				has_country_flag = ISR_independence_secured
			}
		}
		OR = {
			ISR = {
				has_war_with = SYR
				controls_state = 455
				controls_state = 1003
				controls_state = 1004
			}
			ISR = {
				has_war_with = SNP
				controls_state = 455
				controls_state = 1003
				controls_state = 1004
			}
			ISR = {
				has_war_with = CCD
				controls_state = 455
				controls_state = 1003
				controls_state = 1004
			}
		}
    }

	immediate = {
		hidden_effect = {
			ISR = {
				if = {
					limit = {
						has_war_with = SYR
					}
					white_peace = {
						tag = SYR
						message = ISR_syria_peace_tt
					}
				}
				if = {
					limit = {
						has_war_with = SNP
					}
					white_peace = {
						tag = SNP
						message = ISR_syria_peace_tt
					}
				}
				if = {
					limit = {
						has_war_with = CCD
					}
					white_peace = {
						tag = CCD
						message = ISR_syria_peace_tt
					}
				}
				set_state_owner = 454
				set_state_controller = 454
				set_state_owner = 1004
				set_state_controller = 1004
				set_country_flag = ISR_independence_secured
				remove_ideas = SYR_syrian_circlejerk_war
			}
			news_event = {
				id = news.352
			}
			set_global_flag = syr_first_syrian_peace
			1004 = {
				remove_core_of = SYR
				remove_core_of = SNP
				remove_core_of = CCD
				set_state_name = ISR_event_ISR_victory_state_1004
			}
			454 = {
				remove_core_of = SYR
				remove_core_of = SNP
				remove_core_of = CCD
			}
			455 = {
				owner = {
					add_state_claim = 1004
					add_state_claim = 454
				}
			}
		}
	}
	option = {
		name = ISR.victory.a
	}
}
