	trigger = {
		AND = {
			OR = {
				TAG = MAG
				TAG = MLG
			}
			has_war = no
		}
	}
	is_triggered_only = no
	fire_only_once = yes
	option = {
		name = mag.1898.a
		hidden_effect = {
			if = {
				limit = {
					AND = {
						TAG = MAG
						has_war = no
					}
				}
				news_event = {
					id = news.22
				}
				country_event = {
					id = mag.3
					days = 365
				}
				every_state = {
					limit = {
						is_core_of = MLG
					}
					add_core_of = MAG
					remove_core_of = MLG
				}
			}
			else_if = {
				limit = {
					AND = {
						TAG = MLG
						has_war = no
					}
				}
				news_event = {
					id = news.23
				}
				country_event = {
					id = mlg.2
					days = 365
				}
				every_state = {
					limit = {
						is_core_of = MAG
					}
					add_core_of = MLG
				}
			}
		}
	}
}
